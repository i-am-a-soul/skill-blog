### `JSON.stringify + JSON.parse`

```js
let newObj = JSON.parse(JSON.stringify(oldObj))
```

注意：

```js
// ----- 1. -----
const oldObj = {
    a: new Date(), // 字符串
    b: /\.css$/, // {}
    c: new Error('Whoops!'), // {}
    d: (val) => val, // 丢失
    e: Symbol('foo'), // 丢失
    f: undefined, // 丢失
    g: null,
    h: NaN, // null
    i: Infinity, // null
}
console.log(oldObj)
```

```js
// ----- 2. -----
function foo (name) {
    this.name = name
}

const oldObj = new foo('Lucas')
console.log(oldObj.__proto__ === foo.prototype) // true

let newObj = JSON.parse(JSON.stringify(oldObj))
console.log(newObj.__proto__ === foo.prototype) // false

// ----- 3. -----
const oldObj = {}
oldObj.ptr = oldObj // 循环引用

// TypeError: Converting circular structure to JSON
let newObj = JSON.parse(JSON.stringify(oldObj))

// ----- 4. -----
const obj = {
    name: 'Lucas',
}

const oldObj = {
    a: obj,
    b: obj,
}

let newObj = JSON.parse(JSON.stringify(oldObj))

oldObj.a.name = 'foo'
newObj.a.name = 'foo'

console.log(oldObj)
console.log(newObj)
```

### `deepCopy`

```js
function deepCopy (original) {
    const cache = []

    function _deepCopy (original) {
        if (typeof original !== 'object' || original === null) { return original }

        for (let i = 0; i < cache.length; ++ i) {
            if (cache[i].original === original) {
                return cache[i].copy
            }
        }

        const copy = Array.isArray(original) ? [] : {}
        cache.push({ original, copy })
        Object.keys(original).forEach(key => {
            copy[key] = _deepCopy(original[key])
        })
        return copy
    }
    return _deepCopy(original)
}
```
