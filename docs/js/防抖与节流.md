# 防抖与节流

## 防抖

![](/skill-blog/img/0008.webp)

```js
function debounce (func, delay) {
    let timer = null

    return function () {
        if (timer) { clearTimeout(timer) }

        timer = setTimeout(() => {
            func.apply(this, arguments)
        }, delay)
    }
}

function debounce (func, delay, triggerNow) {
    let timer = null

    return function () {
        if (timer) { clearTimeout(timer) }

        if (triggerNow) {
            let firstClick = !timer

            if (firstClick) { func.apply(this, arguments) }

            timer = setTimeout(() => {
                timer = null
            }, delay)
        } else {
            timer = setTimeout(() => {
                func.apply(this, arguments)
            }, delay)
        }
    }
}
```

## 节流

![](/skill-blog/img/0009.webp)

```js
function throttle (func, delay) {
    let prev = 0
    
    return function () {
        const cur = Date.now()
        if (cur - prev > delay) {
            prev = cur
            return func.apply(this, arguments)
        }
    }
}
```
